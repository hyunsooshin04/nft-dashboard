# 코사인 유사도 계산
def
calculate_similarity
(features1, features2):
features1 = features1 / np.linalg.norm(features1)
features2 = features2 / np.linalg.norm(features2)
return
np.dot(features1, features2)

R 코드 예시 - 트렌드 차트 생성

# 메인 트렌드 차트 생성
main_plot <- ggplot(data, aes(x = date, y = mint_count)) +
geom_line(color = "#3b82f6", size = 1.2, alpha = 0.8) +
geom_point(color = "#1e40af", size = 2, alpha = 0.6) +
geom_smooth(method = "loess", se = TRUE, color = "#ef4444") +
labs(title = "NFT Minting Activity Over Time") +
theme_minimal() +
scale_x_date(date_labels = "%m/%d", date_breaks = "1 week")
# PNG 파일로 저장
ggsave("public/charts/minting_trends.png", plot = main_plot, width = 14, height = 8)
